    class Rectangle
    {
        public double width;
        public double length;
        public double Width
        {
            get
            { return width; }
            set
            {
                width = value;
            }
        }
        public double Length
        {
            get
            { return length ; }
            set
            {
                length  = value;
            }
        }

        Rectangle() { }

        public Rectangle(double width=0, double l=0)
        {
            this.width = width;
            length = l;
        }

        public double GetArea(double width, double length)
        {
            return length * width;
        }
        public double GetPerimeter(double width, double length)
        {
            return (length + width) * 2;
        }
        public void display()
        {
            Console.Write("长：");
            Console.WriteLine(this.Length);
            Console.Write("宽：", this.Width);
            Console.WriteLine( this.Width);
            Console.Write("周长：");
            Console.WriteLine(GetPerimeter(this.length, this.width));
            Console.WriteLine("面积：" +GetArea(this.length, this.width));
        }

    }
